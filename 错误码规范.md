# 乐伴好眠后端 API 错误码规范

## 概述

本文档定义了乐伴好眠小程序后端 API 的统一错误码规范，用于规范化接口错误返回，便于前端处理和错误追踪。

---

## 响应格式

### 成功响应

```json
{
  "code": 0,
  "message": "操作成功",
  "data": { ... },
  "timestamp": 1714464000000
}
```

### 错误响应

```json
{
  "code": 1001,
  "message": "参数错误: 用户ID不能为空",
  "data": null,
  "timestamp": 1714464000000
}
```

| HTTP 状态码 | 含义 | 使用场景 |
|------------|------|----------|
| 200 | OK | 请求成功 |
| 400 | Bad Request | 请求参数错误 |
| 401 | Unauthorized | 未授权/登录失效 |
| 402 | Payment Required | 积分不足 |
| 403 | Forbidden | 权限不足 |

| 409 | Conflict | 资源冲突（如重复操作） |

| 502 | Bad Gateway | 外部服务错误 |
| 503 | Service Unavailable | 服务暂时不可用 |

## 使用示例

### 1. 基础用法

```javascript
const { sendSuccess, sendError, ErrorCode } = require('./error-codes');

// 成功响应
app.get('/api/users/:id', (req, res) => {
    const user = findUser(req.params.id);
    if (!user) {
        return sendError(res, ErrorCode.USER_NOT_FOUND);
    }
    sendSuccess(res, user);
});
```

### 2. 带额外信息的错误

```javascript
app.post('/api/posts', (req, res) => {
    const { content } = req.body;
    if (!content) {
        return sendError(res, ErrorCode.MISSING_REQUIRED_PARAM, '帖子内容不能为空');
    }
    if (content.length > 2000) {
        return sendError(res, ErrorCode.CONTENT_TOO_LONG, '帖子内容不能超过2000字');
    }
    // ...
});
```

### 3. 数据库错误转换

```javascript
const { convertDbError, sendError } = require('./error-codes');

try {
    db.prepare('INSERT INTO users (id, email) VALUES (?, ?)').run(id, email);
} catch (err) {
    // 自动转换SQLite错误为标准错误码
    const errorCode = convertDbError(err);
    return sendError(res, errorCode, err.message);
}
```

## 前端处理建议

### 错误码分类处理

```javascript
async function apiCall() {
    try {
        const response = await fetch('/api/...');
        const data = await response.json();

        if (data.code !== 0) {
            handleError(data.code, data.message);
        }

        return data.data;
    } catch (error) {
        console.error('网络错误:', error);
    }
}

function handleError(code, message) {
    // 根据错误码范围处理
    if (code >= 2000 && code < 3000) {
        // 用户相关错误，跳转登录页
        if (code === 2003 || code === 2004) {
            redirectToLogin();
        }
    } else if (code >= 4000 && code < 4200) {
        // 内容相关错误，提示用户
        showToast(message);
    } else if (code >= 6000 && code < 7000) {
        // 外部服务错误，降级处理
        showToast('服务繁忙，请稍后再试');
    } else {
        // 其他错误
        showToast(message || '操作失败');
    }
}
```

## 全局错误处理

后端已配置全局错误处理中间件，用于捕获未处理的错误：

```javascript
// 全局错误处理中间件
app.use((err, req, res, next) => {
    logError('全局错误处理', err, {
        url: req.url,
        method: req.method,
        body: req.body,
        query: req.query
    });

    // 根据错误类型返回不同的错误码
    if (err.name === 'ValidationError') {
        return sendError(res, ErrorCode.INVALID_FORMAT, err.message);
    }
    if (err.name === 'UnauthorizedError') {
        return sendError(res, ErrorCode.UNAUTHORIZED);
    }
    if (err.code === 'ECONNABORTED' || err.code === 'ETIMEDOUT') {
        return sendError(res, ErrorCode.EXTERNAL_API_TIMEOUT);
    }

    sendError(res, ErrorCode.INTERNAL_ERROR, err.message);
});

// 404 处理
app.use((req, res) => {
    sendError(res, ErrorCode.UNKNOWN_ERROR, '接口不存在');
});
```
